网络七层协议

网络的七层协议由底层到顶层包括：物理层，数据链路层，网络层，传输层，会话层，表示层，应用层

socket套接字

传输层是实现端到端的通信,因此，每一个传输层连接有两个端点。传输层连接的端点叫做套接字（socket）。端口号拼接到IP地址就构成了套接字。所谓套接字，实际上是一个通信端点，每个套接字都有一个套接字序号，包括主机的IP地址与一个16位的主机端口号，即形如（主机IP地址：端口号）。例如，如果IP地址是210.37.145.1，而端口号是23，那么得到套接字就是（210.37.145.1：23）

基本操作

#打印设备名称和IPV4地址
import socket
from binascii import hexlify
def get_computer_info():
    host_name=socket.gethostname()
    print(host_name)
    ip_address=socket.gethostbyname(host_name)
    print(ip_address)
#打印远程设备的IP
def get_remote_info():
    remote_host='www.baidu.com'
    try:
        ip_address=socket.gethostbyname(remote_host)
    except socket.error as e:
        print(e)
    print('%s:%s'%(remote_host,ip_address))
#把IPV4地址转换成不同格式
#如果要使用低层网络函数，有时普通的字符串形式的IP地址并不是很有用，需要把它们转换成打包后的32位二进制格式。
def convert_ipv4_address():
    for ip_address in ['127.0.0.1','192.168.1.1']:
        #inet_aton将地址转换为32位的网络序列IP地址
        ip=socket.inet_aton(ip_address)
        print(ip)
        #将二进制IP变为十六进制
        ip16=hexlify(ip)
        print(ip16)
        #inet_ntoa与inet_aton相反
        f_ip=socket.inet_ntoa(ip)
        print(f_ip)
def find_server_name():
    for name in ['http','smtp']:
        get_port=socket.getservbyname(name,'tcp')
        print(get_port)
    print(socket.getservbyname('domain','udp'))
    for port in [80,25]:
        get_name=socket.getservbyport(port,'tcp')
        print(get_name)
    print(socket.getservbyport(53,'udp'))
#Python的 socket 库提供了将数据在网络字节序和主机字节序之间相互转换的函数。ntohl()和htonl()
def convert_integer():
    data=1234
    #Long host byte order,stort host byte order
    print(socket.ntohl(data),socket.ntohs(data))
    #Long Network byte order,short Network byte order
    print(socket.htonl(data),socket.htons(data))
def test_socket_timeout():
    s=socket.socket(socket.AF_INET,socket.SOCK_STREAM)
    print(s.gettimeout())
    s.settimeout(100)
    print(s.gettimeout())
if __name__ == '__main__':
    print("打印设备名称和IPV4地址")
    get_computer_info()
    print("打印远程设备的IP")
    get_remote_info()
    print("把IPV4地址转换成不同格式")
    convert_ipv4_address()
    print("通过指定的端口和协议找到服务名")
    find_server_name()
    print("主机字节序和网络字节序之间相互转换")
    convert_integer()
    print("设定并获取默认的套接字超时时间")
    test_socket_timeout()
