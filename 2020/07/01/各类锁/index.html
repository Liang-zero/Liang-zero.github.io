自旋锁

public class BaseStudy03 {
    AtomicReference<Thread> atomicReference = new AtomicReference<>();
    public void myLock(){
//        获取当前线程
        Thread thread = Thread.currentThread();
        System.out.println(Thread.currentThread().getName()+":"+"come in");
//        当前线程不为空则继续自旋判断
        while (!atomicReference.compareAndSet(null,thread)){
//            System.out.println(Thread.currentThread().getName()+":"+"正在尝试获取锁");
        }
    }

    public void myUnlock(){
//       获取当前线程
        Thread thread = Thread.currentThread();
//        把当前线程设置为空
        atomicReference.compareAndSet(thread,null);
        System.out.println(Thread.currentThread().getName()+":"+"myUnlock");
    }
    public static void main(String[] args)  {
        BaseStudy03 baseStudy03 = new BaseStudy03();
        new Thread(()->{
            baseStudy03.myLock();
           try { TimeUnit.SECONDS.sleep(5); }catch (InterruptedException e){e.printStackTrace();}
           baseStudy03.myUnlock();
        },"t1").start();

       try { TimeUnit.SECONDS.sleep(1); }catch (InterruptedException e){e.printStackTrace();}


        new Thread(() -> {
            baseStudy03.myLock();
            baseStudy03.myUnlock();
        }, "t2").start();

    }
}

读写锁

package com.study;

import java.util.HashMap;
import java.util.Map;
import java.util.concurrent.TimeUnit;
import java.util.concurrent.locks.ReentrantReadWriteLock;

class MyCache{
//    修改数据其他线程可见
    private volatile Map<String,Object> map = new HashMap<>();
//    读写锁或者共享锁
    private ReentrantReadWriteLock rwLock = new ReentrantReadWriteLock();

    public void put(String key, Object value){
//        读锁
        rwLock.writeLock().lock();
        try {
            System.out.println("正在写入:"+key);
            try { TimeUnit.MILLISECONDS.sleep(300); }catch (InterruptedException e){e.printStackTrace();}
            map.put(key,value);
            System.out.println(key+"写入完成");
        }catch (Exception e){
            e.printStackTrace();
        }finally {
//            释放读锁
            rwLock.writeLock().unlock();
        }

    }
    public void get(String key){
//        写锁
        rwLock.readLock().lock();
        try {
            System.out.println("正在读取:"+key);
            try { TimeUnit.MILLISECONDS.sleep(300); }catch (InterruptedException e){e.printStackTrace();}
            map.get(key);
            System.out.println(key+"读取完成");
        }catch (Exception e){
            e.printStackTrace();
        }finally {
//            释放写锁
            rwLock.readLock().unlock();
        }
    }
}


public class BaseStudy04 {
    public static void main(String[] args)  {

        MyCache cache = new MyCache();
//        创建五个线程用来写
        for (int i = 0; i <5; i++) {
            final int tmp = i;
            new Thread(()->{
                   cache.put(tmp+"",tmp+"");
                },String.valueOf(i)).start();
        }
//          创建五个线程用来读
        for (int i = 0; i <5; i++) {
            final int tmp = i;
            new Thread(()->{
                cache.get(tmp+"");
            },String.valueOf(i)).start();
        }
    }
}

CountDownLatch/CyclicBarrier/Semaphore

CountDownLatch

一个同步辅助类，在完成一组正在其他线程中执行的操作之前，它允许一个或多个线程一直等待。 

public class BaseStudy05 {
    public static void main(String[] args) throws InterruptedException {
        CountDownLatch downLatch = new CountDownLatch(6);
        for (int i = 0; i <6; i++) {
            final int temp = i;
            new Thread(()->{
                System.out.println("同学"+temp+"离开");
                downLatch.countDown();
                },String.valueOf(i)).start();
        }
        downLatch.await();
        System.out.println("班长离开");
    }
}

CyclicBarrier

一个同步辅助类，它允许一组线程互相等待，直到到达某个公共屏障点 (common barrier point)。在涉及一组固定大小的线程的程序中，这些线程必须不时地互相等待，此时 CyclicBarrier 很有用。因为该 barrier 在释放等待线程后可以重用，所以称它为循环 的 barrier。 

public class BaseStudy06 {
    public static void main(String[] args)  {
        CyclicBarrier barrier = new CyclicBarrier(7,()->{
            System.out.println("召唤神龙");
        });
        for (int i = 0; i < 7; i++) {
            final int temp = i;
            new Thread(()->{
                System.out.println("收集第"+temp+"颗龙珠");
                try {
                    barrier.await();
                } catch (InterruptedException e) {
                    e.printStackTrace();
                } catch (BrokenBarrierException e) {
                    e.printStackTrace();
                }
            },String.valueOf(i)).start();
        }


    }
}

Semaphore

一个计数信号量。从概念上讲，信号量维护了一个许可集。如有必要，在许可可用前会阻塞每一个 acquire()，然后再获取该许可。每个 release() 添加一个许可，从而可能释放一个正在阻塞的获取者。但是，不使用实际的许可对象，Semaphore 只对可用许可的号码进行计数，并采取相应的行动。 

public class BaseStudy07 {
    public static void main(String[] args)  {
        Semaphore semaphore = new Semaphore(3);
        for (int i = 0; i < 6; i++) {
            new Thread(()->{
                try {
                    semaphore.acquire();
                    System.out.println(Thread.currentThread().getName()+"抢到车位");
                    try { TimeUnit.SECONDS.sleep(3); }catch (InterruptedException e){e.printStackTrace();}
                    System.out.println(Thread.currentThread().getName()+"停车三秒");
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }finally {
                    semaphore.release();
                }
            },String.valueOf(i)).start();
        }
    }
}
