django之文件上传

方法一

创建一个static文件，在static下创建一个img文件存放图片

setting.py

STATICFILES_DIRS = [
    os.path.join(BASE_DIR, 'static')
]

urls.py

urlpatterns = [
    url(r'^upload_1', views.upload_1, name='upload_1')
]

views.py

def upload_1(request):
    if request.method == "GET":
        return render(request, 'upload_1.html')
    elif request.method == "POST":
        # 获取文件用FILES
        icon = request.FILES.get('icon')
        with open('static/img/2.jpg', 'wb') as f:
            # 上传时时一块一块的上传，所以用for
            for part in icon.chunks():
                f.write(part)
        return HttpResponse('success!!!')

upload.html

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
</head>
<body>
<form action="{% url 'one:upload_1' %}" method="post" enctype="multipart/form-data">
    {% csrf_token %}
    <span>文件</span><input type="file" name="icon" placeholder="请选择文件">
    <button>上传</button>
</form>
</body>
</html>

方法二

将文件路径存储在数据库中

setting.py

# 在setting.py中添加
MEDIA_ROOT = os.path.join(BASE_DIR, 'static/upload')

models.py

class UserModel(models.Model):
    name = models.CharField(max_length=20)
    # uplad_to指定文件路径
    icon = models.ImageField(upload_to='icons')

urls.py

urlpatterns = [
    url(r'^upload_1', views.upload_1, name='upload_1'),
    url(r'^upload_2', views.upload_2, name='upload_2'),
]

views.py

def upload_2(request):
    if request.method == "GET":
        return render(request, 'upload_2.html')
    elif request.method == "POST":
        icon = request.FILES.get('icon')
        name = request.POST.get('username')
        user = UserModel()
        user.icon = icon
        user.name = name
        user.save()
        return HttpResponse('Success!!')

开启服务器,访问

在网页显示用户名头像

views.py

def get_icon(request):
    username = request.GET.get('username')
    user = UserModel.objects.get(name=username)
    # 这里的url是MEDIA_ROOT+存储在数据的路径
    icon_url = '/static/upload/' + user.icon.url
    return render(request, 'show.html', locals())

show.hml

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
</head>
<body>
    <li>{{ username }}</li>
    <img src="{{ icon_url }}" alt="头像">
</body>
</html>

注意

在linux系统中有一个小bug，当一个文件夹的文件超过65535时，文件就打不开，为了避免这种情况，在models.py中设置

class UserModel(models.Model):
    name = models.CharField(max_length=20)
    # 用年月日分层级作为文件夹，这样就可以有效的避免这个bug
    icon = models.ImageField(upload_to='%Y/%m/%d/icons')
